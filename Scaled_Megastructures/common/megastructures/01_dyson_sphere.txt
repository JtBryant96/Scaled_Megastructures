dyson_sphere_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	place_entity_on_planet_plane = no
	entity_offset = { x = -7 y = -7 }
	build_time = 1800
	build_cost = {
		minerals = 1000
		influence = 250
	}
	construction_blocks_others = yes
	maintenance = {
		energy = 5
	}

	potential = {
		has_ascension_perk = ap_galactic_wonders
	}
	
	possible = {
		from = {
			has_technology = tech_mega_engineering
		}
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_planet = {
					is_surveyed = {
						who = prev.from

					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_black_hole_neutron_star_pulsar"
			NOT = {
				is_star_class = sc_black_hole

				is_star_class = sc_pulsar
			}
		}
		custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				NOR = {
					has_star_flag = dyson_sphere_built
				}
		}
	}
	
	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "must_build_around_star"
				is_star = yes
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "requires_no_orbital_station"
				has_orbital_station = no
			}
		}
	}
	
	ai_weight = {
		factor = 0
	}
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		set_star_flag = dyson_sphere_built
		if = {
			limit = {
				is_star_class = sc_b
			}
			remove_megastructure = fromfrom
			spawn_megastructure = {
				type = "dyson_sphere_b_0"
				owner = from
				graphical_culture = fromfrom
				location = planet
			}
		}
		if = {
			limit = {
				is_star_class = sc_a
			}
			remove_megastructure = fromfrom
			spawn_megastructure = {
				type = "dyson_sphere_a_0"
				owner = from
				graphical_culture = fromfrom
				location = planet
			}
		}
		if = {
			limit = {
				is_star_class = sc_f
			}
			remove_megastructure = fromfrom
			spawn_megastructure = {
				type = "dyson_sphere_f_0"
				owner = from
				graphical_culture = fromfrom
				location = planet
			}
		}
		if = {
			limit = {
				is_star_class = sc_g
			}
			remove_megastructure = fromfrom
			spawn_megastructure = {
				type = "dyson_sphere_g_0"
				owner = from
				graphical_culture = fromfrom
				location = planet
			}
		}
		if = {
			limit = {

			}

# Restored Dyson Sphere
dyson_sphere_restored = {
	entity = "dyson_sphere_phase_05_entity"
	construction_entity = "dyson_sphere_part_4_entity"
	portrait = "GFX_megastructure_dyson_sphere_background"
	upgrade_from = {
		dyson_sphere_ruined
	}
	build_time = 7200
	build_cost = {
		minerals = 50000
	}
	monthly_production = {
		energy = 1000
	}

	possible = {
		from = { has_technology = tech_mega_engineering }		
	}

	on_build_start = {
		fromfrom = {
			set_graphical_culture = root.from
		}
	}

	on_build_complete = {
		every_system_planet = {
			limit = {
				OR = {
					is_planet_class = pc_molten
					is_planet_class = pc_toxic
				}
			}
			change_pc = pc_frozen
		}
		every_system_planet = {
			limit = {
				is_planet_class = pc_barren
			}
			change_pc = pc_barren_cold
		}
		every_system_ambient_object = { destroy_ambient_object = this }
		if = {
			limit = {
				is_star_class = sc_g
			}
			remove_megastructure = fromfrom
			spawn_megastructure = {
				type = "dyson_sphere_g_5"
				owner = from
				graphical_culture = fromfrom
				location = planet
			}
		}
		from = {
			country_event = { id = utopia.2011 }
		}
	}
}
